Creating custom parsers and formatters:

  • Validating values and propagating to form: 
                    function validate(value){
                    var isValid = validateFn(value);
                    ngModelCtrl.$setValidity('validationName', isValid);
                    return isValid;
                }
  • Apply in parsers and formatters: 
                    ngModelCtrl.$parsers.unshift(function(value) {
                    return validate(value) ? value : undefined;
                });

                ngModelCtrl.$formatters.unshift(function(value) {
                    validate(value);
                    return value;
                });
