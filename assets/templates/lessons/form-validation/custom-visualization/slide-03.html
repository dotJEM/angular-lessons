Advanced visualization with directives:

app.directive('dxErrorPopover', function($parse) {
    return {
        restrict: 'A',
        require: 'ngModel',
        compile: function(elm, attr) {
        var textGetter = $parse(attr.dxErrorPopover);

            return function(scope, elem, attr, ctrl) {
                function showIfInvalid(){
                  if(ctrl.$invalid){ element.popover('show'); }
                  else { element.popover('hide'); }
                }

                element.focus(function(){
                  showIfInvalid();
                });

                element.blur(function(){
                  element.popover('hide');
                });

                element.popover({
                  content: textGetter(scope),
                  placement: 'top',
                  trigger: 'manual'
                });
            }
        }
    };
});
